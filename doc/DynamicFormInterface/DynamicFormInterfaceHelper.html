def find_person_i18n(field_title)
  if field_title.to_s.blank? # avoiding "".to_sym will caused interning empty string error
    return ""
  elsif field_title =~ /person./ # person.male, person.famale...etc person build-in select options
    return field_title.t
  elsif field_title
    return field_title
  else
    return ""
  end
end

def render_edit_input( target, field, options = {} )
  options.merge!( :disabled => "disabled" )
  input = "#{render_label(target,field, options)} #{send( "render_input_#{field_type(field)}", target, field, options )}"

  if field.new_record? 
    #build-in
    if field.attr_name == :email
      "<div id=\"form-field-#{field.attr_name}\"> #{input} </div>" 
      #TODO: now this depend on EventForm   
    elsif @event.form.respond_to? field.attr_name 
      "<div id=\"form-field-#{field.attr_name}\"> #{input} #{require_field field.attr_name,@event} #{disable_field field.attr_name} </div>" 
    end
  else
    # custom field
    "<div id=\"form-field-#{field.attr_name}\"> #{input} #{require_custom_field field} #{remove_custom_field field.id} #{edit_custom_field field.id} #{move_higher_custom_field field.id} #{move_lower_custom_field field.id}</div>"
  end
end

def render_input_field_attachment( target, field, options = {})
  tmp = ''
  if options[:object] && !options[:object].new_record?
    tmp = render_field_attachment(options[:object],field)+ " , "
  end
  
  object_name=target.class.to_s.underscore
  value = (target.custom_file_values.select{|x| x.form_field_id == field.id } ).first

  field_basename = "#{object_name}[#{field.values_name}_attributes][#{field.id}]"
  tmp += file_field_tag("#{field_basename}[asset]", options)
  tmp += hidden_field_tag("#{field_basename}[form_field_id]", field.id )
  tmp += hidden_field_tag("#{field_basename}[id]", value.id ) if value && value.id
  return tmp

end

def render_input_field_checkbox( target, field, options = {} )
  tmp = ''
  object_name=target.class.to_s.underscore
  values = target.custom_option_values.select{|x| x.custom_option && x.custom_option.form_field_id == field.id }
  
  if target.id 
    values.each do |v|
      tmp += hidden_field_tag("#{object_name}[#{field.values_name}_attributes][del#{v.id}][id]", v.id)
      tmp += hidden_field_tag("#{object_name}[#{field.values_name}_attributes][del#{v.id}][_destroy]", true)
    end
  end

  field.custom_options.each do |o|
    value = values.find{|x| x.custom_option_id == o.id }
    i18n_label = find_person_i18n(o.name)
    field_name = "#{object_name}[#{field.values_name}_attributes][#{field.id}_#{o.id}][custom_option_id]"
    tmp += check_box_tag( field_name, o.id, value )
    tmp += label_tag( field_name, i18n_label, :class=> "inline")
    tmp += text_field_tag( "#{field_name}[comment]", value && value.comment) if o.commentable
    tmp += "<br>".html_safe
  end


  return tmp
end

def render_label( target, field, options = {} )

  i18n_label = find_person_i18n(field.title)
  el = target.is_a?(String) ? target : target.class.to_s

  if field.is_required
    "<label for=\"#{el}_#{field.id}\" class=\"desc\"><span class=\"require\">*</span>#{i18n_label}</label>"
  else 
    "<label for=\"#{el}_#{field.id}\" class=\"optional desc\">#{i18n_label}</label>"
  end
end

def render_search_input(field, value )
  content_tag :li do
    raw(render_label("search",field)) + 
       raw( content_tag( :div, :id=>"search_#{field.id}",:class=>"col") do
        if field.kind_of?(ColumnOption)
          raw( field.custom_options.map do |option|
            raw(check_box_tag( "search[#{field.id}][]",option.id, value && value.include?(option.id.to_s)) + label_tag(option.name))
          end.join)
        else
          raw text_field_tag "search[#{field.id}]", value
        end
       end)
  end
end
