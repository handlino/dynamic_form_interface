def set_relaction
  self.form_type = self.applicant.form.class.to_s
  self.form_id = self.applicant.form.id
  if self.form_id != self.form_field.form_id
    errors.add("form_field_#{form_field.id}", I18n.t("errors.messages.invalid") )
    return false
  end
end
